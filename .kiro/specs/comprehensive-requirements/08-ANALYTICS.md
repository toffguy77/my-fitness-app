# Аналитика и Прогресс

## Обзор

Раздел **Аналитика** предоставляет пользователю детальное представление о его прогрессе через графики, метрики, тренды и инсайты. Это место, где пользователь видит результаты своих усилий и получает мотивацию продолжать.

---

## Структура Раздела

### Основные Компоненты

1. **Progress Overview** — Обзор прогресса
2. **Body Metrics** — Метрики тела (вес, объемы, жир%)
3. **Nutrition Analytics** — Аналитика питания
4. **Workout Analytics** — Аналитика тренировок
5. **Habits Tracking** — Отслеживание привычек
6. **Photo Progress** — Фото прогресса
7. **Reports** — Отчеты и экспорт данных

```javascript
// Агрегация метрик за период
async function aggregateMetrics(userId, startDate, endDate) {
  const [weight, nutrition, workouts, habits] = await Promise.all([
    db.weight.aggregate(userId, startDate, endDate),
    db.nutrition.aggregate(userId, startDate, endDate),
    db.workouts.aggregate(userId, startDate, endDate),
    db.habits.aggregate(userId, startDate, endDate)
  ]);

  return {
    weight: {
      current: weight.latest,
      start: weight.first,
      change: weight.latest - weight.first,
      changePercent: ((weight.latest - weight.first) / weight.first) * 100,
      average: weight.average,
      trend: calculateTrend(weight.data)
    },
    nutrition: {
      averageCalories: nutrition.avgCalories,
      averageProtein: nutrition.avgProtein,
      averageFats: nutrition.avgFats,
      averageCarbs: nutrition.avgCarbs,
      adherence: nutrition.adherenceRate,
      daysLogged: nutrition.daysLogged
    },
    workouts: {
      total: workouts.count,
      planned: workouts.planned,
      completionRate: (workouts.count / workouts.planned) * 100,
      totalDuration: workouts.totalMinutes,
      avgDuration: workouts.avgMinutes,
      totalCalories: workouts.totalCalories,
      byType: workouts.groupedByType
    },
    habits: {
      water: habits.water.completionRate,
      sleep: habits.sleep.average,
      steps: habits.steps.average
    }
  };
}
```

```javascript
async function generateInsights(userId, metrics) {
  const insights = [];

  // Анализ тренда веса
  if (metrics.weight.trend === 'declining' && metrics.nutrition.adherence > 0.8) {
    insights.push({
      type: 'positive',
      category: 'weight',
      message: 'Вес снижается стабильно при хорошем соблюдении питания'
    });
  }

  // Анализ белка
  if (metrics.nutrition.averageProtein < metrics.goals.protein * 0.9) {
    insights.push({
      type: 'warning',
      category: 'nutrition',
      message: `Потребление белка ниже цели на ${Math.round((1 - metrics.nutrition.averageProtein / metrics.goals.protein) * 100)}%`,
      suggestion: 'Добавь протеиновый коктейль или увеличь порцию белка в обеде'
    });
  }

  // Корреляция сна и тренировок
  const sleepWorkoutCorr = calculateCorrelation(
    metrics.habits.sleep.data,
    metrics.workouts.completionByDay
  );

  if (sleepWorkoutCorr > 0.6) {
    insights.push({
      type: 'insight',
      category: 'habits',
      message: 'Когда вы спите 7+ часов, вероятность тренировки увеличивается на 35%'
    });
  }

  return insights;
}
```
