func checkTablesExist(db *sql.DB) bool {
        tables := []string{
                "coach_client_relationships",
                "daily_metrics",
                "weekly_plans",
                "tasks",
                "weekly_reports",
                "weekly_photos",
        }

        for _, table := range tables {
                var exists bool
                err := db.QueryRow(`
                        SELECT EXISTS (
                                SELECT FROM information_schema.tables
                                WHERE table_name = $1
                        )
                `, table).Scan(&exists)
                if err != nil {
                        log.Printf("Warning: Failed to check table %s: %v", table, err)
                        continue
                }
                if exists {
                        return true
                }
        }
        return false
}

func runMigrationFile(dbURL, filePath string) error {
        // Get absolute path
        absPath, err := filepath.Abs(filePath)
        if err != nil {
                return fmt.Errorf("failed to get absolute path: %v", err)
        }

        // Check if file exists
        if _, err := os.Stat(absPath); os.IsNotExist(err) {
                return fmt.Errorf("migration file not found: %s", absPath)
        }

        // Read migration file
        content, err := os.ReadFile(absPath)
        if err != nil {
                return fmt.Errorf("failed to read migration file: %v", err)
        }

        // Connect and execute
        db, err := sql.Open("postgres", dbURL)
        if err != nil {
                return fmt.Errorf("failed to connect to database: %v", err)
        }
        defer db.Close()

        // Execute migration
        _, err = db.Exec(string(content))
        if err != nil {
                return fmt.Errorf("failed to execute migration: %v", err)
        }

        return nil
}

func verifyRLSPolicies(db *sql.DB) {
        tables := []string{
                "coach_client_relationships",
                "daily_metrics",
                "weekly_plans",
                "tasks",
                "weekly_reports",
                "weekly_photos",
        }

        fmt.Println("\nRLS enabled on tables:")
        for _, table := range tables {
                var rlsEnabled bool
                err := db.QueryRow(`
                        SELECT rowsecurity
                        FROM pg_tables
                        WHERE schemaname = 'public' AND tablename = $1
                `, table).Scan(&rlsEnabled)
                if err != nil {
                        fmt.Printf("  ✗ %s (error checking: %v)\n", table, err)
                        continue
                }
                if rlsEnabled {
                        // Count policies
                        var policyCount int
                        db.QueryRow(`
                                SELECT COUNT(*)
                                FROM pg_policies
                                WHERE tablename = $1
                        `, table).Scan(&policyCount)
                        fmt.Printf("  ✓ %s (%d policies)\n", table, policyCount)
                } else {
                        fmt.Printf("  ✗ %s (RLS not enabled)\n", table)
                }
        }
}

        expectedIndexes := map[string][]string{

                },
        }

        fmt.Println("\nIndexes created:")
        totalIndexes := 0
        for table, indexes := range expectedIndexes {
                for _, indexName := range indexes {
                        var exists bool
                        err := db.QueryRow(`
                                SELECT EXISTS (
                                        SELECT FROM pg_indexes
                                        WHERE tablename = $1 AND indexname = $2
                                )
                        `, table, indexName).Scan(&exists)
                        if err != nil {
                                fmt.Printf("  ✗ %s.%s (error checking: %v)\n", table, indexName, err)
                                continue
                        }
                        if exists {
                                fmt.Printf("  ✓ %s.%s\n", table, indexName)
                                totalIndexes++
                        } else {
                                fmt.Printf("  ✗ %s.%s (not found)\n", table, indexName)
                        }
                }
        }
        fmt.Printf("\nTotal indexes created: %d\n", totalIndexes)
}

func loadEnv(filename string) error {
        file, err := os.Open(filename)
        if err != nil {
                return err
        }
        defer file.Close()

        // Simple .env parser
        var lines []string
        content, err := os.ReadFile(filename)
        if err != nil {
                return err
        }

        for _, line := range strings.Split(string(content), "\n") {
                line = strings.TrimSpace(line)
                if line == "" || strings.HasPrefix(line, "#") {
                        continue
                }
                lines = append(lines, line)
        }

        for _, line := range lines {
                parts := strings.SplitN(line, "=", 2)
                if len(parts) != 2 {
                        continue
                }
                key := strings.TrimSpace(parts[0])
                value := strings.TrimSpace(parts[1])
                os.Setenv(key, value)
        }

        return nil
}
